function init() { // Fetch all historical answers from the database fetchHistoricalAnswers().then(historicalAnswers => { const askedQuestions = new Set(historicalAnswers.map(answer => answer.question)); // Select new questions only from the remaining pool for each category const newQuestions = { tot: filterQuestionsByCategory('tot', askedQuestions), ranking: filterQuestionsByCategory('ranking', askedQuestions), text: filterQuestionsByCategory('text', askedQuestions) }; // If all questions in a category have been used, fall back to random selection for (const category in newQuestions) { if (newQuestions[category].length === 0) { newQuestions[category] = getRandomQuestions(category); } } // Proceed with selected new questions startQuiz(newQuestions); }); } function filterQuestionsByCategory(category, askedQuestions) { // Assume getQuestionsByCategory is a function that returns all questions for a given category return getQuestionsByCategory(category).filter(question => !askedQuestions.has(question)); } function getRandomQuestions(category) { // Assume getAllQuestions is a function that returns all available questions for the category return getAllQuestions(category).sort(() => Math.random() - 0.5); }
